Destes, os parâmetros do circuito equivalente são calculados em \cref{eq:parametros_circuito_equivalente}.

\begin{equation}
    \begin{aligned}
        w_{e} & = 2 \pi f_{n} = 2 \pi \cdot 60 = 376.99 \text{ rad/s} \\
        V_{ph} & = \frac{V_{n}}{\sqrt{3}} = \frac{460}{\sqrt{3}} = 265.58 \text{ V} \\
        R_{s} & = 0.0302 \Omega \\
        X_{s} & = w_{e} L_{s} = 376.99 \cdot 0.000283 = 0.1067 \Omega \\
        X_{m} & = w_{e} L_{m} = 376.99 \cdot 0.01095 = 4.126 \Omega \\
        X_{r}' & = w_{e} L_{r}' = 376.99 \cdot 0.000283 = 0.1067 \Omega \\
        R_{r} & = \frac{R_{r}'}{s} = \frac{0.1721}{s} \Omega \\
    \end{aligned}
    \label{eq:parametros_circuito_equivalente}
\end{equation}
